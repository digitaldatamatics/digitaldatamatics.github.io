const levels = { 
  error: 0,
  warn: 1,
  info: 2,
  http: 3,
  verbose: 4,
  debug: 5,
  silly: 6
};


const winston = require('winston');

const logger = winston.createLogger({
  level: 'info', //default
  format: winston.format.json(),
  defaultMeta: { service: 'user-service' },
  transports: [
    new winston.transports.File({ filename: 'error.log', level: 'error' }),  // logs with level `error` and below
    new winston.transports.File({ filename: 'combined.log' }), //logs with level `info`  (as per default)
  ],
});

// Other than exceptionHandlers, Alternatively you can set `handleExceptions` to true when adding transports

winston.add(new winston.transports.File({
    filename: 'path/to/combined.log',
    handleExceptions: true
}));
