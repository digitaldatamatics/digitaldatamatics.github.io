Importâ€‹

import {
  setLogger,

Pass configâ€‹

simple db log exampleâ€‹

//syntax
setLogger(options);

setLogger({
//by default console & file logs are enabled, so enable db logs    
enableDatabaseLog: true,
databaseOptions: {
    connectionURL: "{DB_CONNECTION_URL}",
    collectionName: "Logs",
},

applicationName: "{APPLICATION_NAME}",
logFolderPathPrefix:  "./logs/",
});

  • Collection is auto created but db must exists!

full log exampleâ€‹

//example
setLogger({

//console
enableConsoleLog: true,
consoleLogLevel: LOG_LEVEL_NAME.INFO,

//file
enableFileLog: true,
fileLogLevel: LOG_LEVEL_NAME.INFO,

//rotate file log
enableRotateFileLog: true,
rotateFileLogLevel: LOG_LEVEL_NAME.INFO,
rotateFileOptions: {
    maxNumberOfFilesInDays: 60 
},

//database log
enableDatabaseLog: true,
databaseLogLevel: LOG_LEVEL_NAME.ERROR,
databaseOptions: {
    connectionURL: "{DB_CONNECTION_URL}",
    collectionName: "Logs",
    maxNumberOfDocuments: 100000,
    maxSizeOfCollectionInBytes: (1024 * 1024 * 50), //50 MB
},

//aws cloudwatch
enableAwsCloudWatchLog: true,
awsCloudWatchLogLevel: LOG_LEVEL_NAME.INFO,
awsConfig: {
    accessKeyId: "{AWS_ACCESS_KEY_ID}",
    secretAccessKey: "{AWS_SECRET_KEY}",
    region: "{AWS_REGION}",
}
awsCloudWatchOptions: {
    maxNumberOfDays: 60,
    name: "",
    groupName: "",
    streamName: "",
},

//common
applicationName: APPLICATION_NAME,
logFolderPathPrefix:  "./logs/" ,
});
