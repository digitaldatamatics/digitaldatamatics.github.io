Dateâ€‹

convertDateToStringâ€‹


 const todayDateIndianStr = await commonUtils.DateCls.convertDateToString(new Date(), commonUtils.DATE_FORMATS.INDIAN);
//Eg: todayDateIndianStr = 22/11/2021

 const todayDateISOStr = await commonUtils.DateCls.convertDateToString(new Date());
//Eg: todayDateIndianStr = 2021-11-22T12:49:32.503Z (typical API or database format)

convertStringToDateâ€‹

//Pass string & format
 const todayDateInst = await commonUtils.DateCls.convertStringToDate("22/11/2021", commonUtils.DATE_FORMATS.INDIAN);
//Eg: todayDateInst is instance of Date

//Pass ISO string (typical API or database format)
 const todayDateInst2 = await commonUtils.DateCls.convertStringToDate("2021-11-22T12:49:32.503Z");


 //Pass string & format
 const dateTimeStr = "22/11/2021 13:06:07";
 const dateTimeFormat = commonUtils.DATE_FORMATS.INDIAN + " " + commonUtils.TIME_FORMATS.HOUR_24;
 const todayDateInst3 = await commonUtils.DateCls.convertStringToDate(dateTimeStr, dateTimeFormat);

dateDiffâ€‹

const minDate = new Date("2021-10-18");
const maxDate = new Date("2021-11-19");

//difference in days
const checkValueDays = await commonUtils.DateCls.dateDiff(maxDate, minDate, commonUtils.DATE_TIME_UNITS.DAYS, true);

//difference in months
const checkValueMonth = await commonUtils.DateCls.dateDiff(maxDate, minDate, commonUtils.DATE_TIME_UNITS.MONTHS, true);
//difference in months with decimal
const checkValueMonth2 = await commonUtils.DateCls.dateDiff(maxDate, minDate, commonUtils.DATE_TIME_UNITS.MONTHS, false);

addDurationToDateâ€‹

const date = new Date("2021-10-18");
const dayDiff = 1;
const monthDiff = 1;
const hourDiff = 2;

const durationDaysObj: DateTypes.IMomentDuration = { days: dayDiff };
const durationMonthsObj: DateTypes.IMomentDuration = { months: monthDiff };
const durationMixedObj: DateTypes.IMomentDuration = { months: monthDiff, days: dayDiff, hours : hourDiff};

//adding days to existing date
const newDate = await commonUtils.DateCls.addDurationToDate(date, durationDaysObj);

//adding months to existing date
const newDate2 = await commonUtils.DateCls.addDurationToDate(date, durationMonthsObj);

//adding multiple to existing date
const newDate3 = await commonUtils.DateCls.addDurationToDate(date, durationMixedObj);

convertNestedStringToDateâ€‹

  • Useful to convert front end API json dates to date objects before inserting in database
const obj = {
        name: "alex",
        dob: "2021-11-22T11:40:35.477Z",
        family: {
            fatherDob: "1972-11-01",
            motherDob: "1970-06-01",
            others: {
                location: "India",
                brotherDob: "1991-09-02"

            }
        }
    }

const checkObj = await commonUtils.DateCls.convertNestedStringToDate(obj);   
/*
 checkObj = {
      name: 'alex',
      dob: 2021-11-22T11:40:35.477Z, //date object
      family: {
        fatherDob: 1972-11-01T00:00:00.000Z,
        motherDob: 1970-06-01T00:00:00.000Z,
        others: { location: 'India', brotherDob: 1991-09-02T00:00:00.000Z }
      }
    }
*/
