tsconfig.jsonâ€‹

{
    "compilerOptions": {
        "module": "commonjs", //node require syntax
        "esModuleInterop": true, //es6 import syntax
        "resolveJsonModule": true,
        "target": "es6",
        "noImplicitAny": true,
        "moduleResolution": "node", //asking ts to use nodejs resolving style for file paths (Point A)
        "sourceMap": true, //for debugging
        "outDir": "dist",
        "baseUrl": ".", //above module resolution base url
        "paths": { //help the compiler to find .d.ts type definitions
            "*": [
                "node_modules/*",
                "src/types/*"
            ]
        }
    },
    "include": [
        "src/**/*"
    ]
}
